# Firebase Security Rules for QR Attendance System
# Copy these rules to Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow read access to all collections for now
    // In production, you should implement proper authentication
    
    // Students collection - allow read/write for now
    match /students/{document} {
      allow read, write: if true;
    }
    
    // Attendance collection - allow read/write for now  
    match /attendance/{document} {
      allow read, write: if true;
    }
    
    // Events collection - allow read/write for now
    match /events/{document} {
      allow read, write: if true;
    }
    
    // For production, replace with proper authentication:
    // match /students/{document} {
    //   allow read, write: if request.auth != null;
    // }
  }
}
